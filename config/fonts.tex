% Configurações de fonte e parágrafos
% Requer XeLaTeX ou LuaLaTeX (fontspec)

\usepackage{fontspec}
\usepackage{xcolor} % para \colorbox (realces no modelo)

% Fonte padrão: Arial 12
\setmainfont{Arial}

% 1. Desativa completamente o corte de palavras com hífen
\usepackage[none]{hyphenat}

% 2. Configurações extremas para forçar o texto a esticar dentro das margens
\sloppy 
\setlength{\emergencystretch}{3em} 
\tolerance=9999 

\usepackage{setspace}
\onehalfspacing

% Recuo de parágrafo ("tab")
\setlength{\parindent}{1.25cm}

% Um espaço de uma linha entre parágrafos
\setlength{\parskip}{6pt}

% Garante recuo também no primeiro parágrafo após títulos (\section, etc.)
\usepackage{indentfirst}
